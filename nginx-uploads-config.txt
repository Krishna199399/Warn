# Add this to /etc/nginx/sites-available/api.warnamayii.cloud.conf
# Inside the "server" block that handles HTTPS (port 443)

    # Serve uploaded files directly (images, documents, etc.)
    location /uploads {
        alias /var/www/Warn/wans-backend/uploads;
        expires 1y;
        add_header Cache-Control "public, immutable";
        add_header Access-Control-Allow-Origin "https://warnamayii.cloud";
        
        # Security headers
        add_header X-Content-Type-Options "nosniff";
        
        # Allow common image formats
        location ~* \.(jpg|jpeg|png|gif|webp|svg)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }

# IMPORTANT: This location block should be BEFORE the "location /api" block
# and BEFORE any catch-all "location /" block
